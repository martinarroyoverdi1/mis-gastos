function setSyncStatus(state) {

  if (!el) return;

  const states = {

    ok:      { text: '● Google Sheets sincronizado', color: '#30D158' },

  };

  el.textContent = s.text;

  el.style.color = s.color;

}


async function sheetPost(payload) {

  const resp = await fetch(SHEET_URL, { method: 'POST', body: JSON.stringify(payload) });

  return resp.text();

}


async function loadFromSheet() {

  setSyncStatus('loading');

  try {

    const resp = await fetch(SHEET_URL + '?action=getAll');

    const rows = await resp.json();

    if (!Array.isArray(rows)) throw new Error('bad response');

    expenses = rows.map(r => ({

      id: Number(r.id), date: r.fecha, desc: r.descripcion,

      cat: r.categoria, amount: Number(r.monto), currency: r.moneda,

    })).sort((a,b) => new Date(b.date) - new Date(a.date));

    saveLocal(); setSyncStatus('ok'); renderAll();

  } catch { setSyncStatus('error'); renderAll(); }

}


async function addExpense() {

  const desc = document.getElementById('desc').value.trim();

  const cat = document.getElementById('cat').value;

  const amount = parseFloat(document.getElementById('amount').value);

  const currency = document.getElementById('currency').value;

  if (!desc) { flash('desc'); return; }

  if (!amount || amount <= 0) { flash('amount'); return; }

  const expense = { id: Date.now(), desc, cat, amount, currency, date: new Date().toISOString() };

  expenses.unshift(expense); saveLocal();

  document.getElementById('desc').value = '';

  document.getElementById('amount').value = '';

  renderAll(); setSyncStatus('syncing');

  try { await sheetPost({ action: 'add', expense }); setSyncStatus('ok'); }

  catch { setSyncStatus('error'); }

}


function flash(id) {

}

  setSyncStatus('syncing');

  try { await sheetPost({ action: 'delete', id }); setSyncStatus('ok'); }

  catch { setSyncStatus('error'); }

}

  const d = new Date(iso), today = new Date(), yest = new Date(today);

  yest.setDate(today.getDate()-1);

  const t = d.toLocaleTimeString('es-AR',{hour:'2-digit',minute:'2-digit'});

  if (d.toDateString() === today.toDateString()) return 'Hoy · ' + t;

  if (d.toDateString() === yest.toDateString()) return 'Ayer · ' + t;

  return d.toLocaleDateString('es-AR',{day:'numeric',month:'short'}) + ' · ' + t;

}

}

function renderDashboard() {

  const r30 = last30(), arsOnly = r30.filter(e => e.currency === 'ARS');

  document.getElementById('dash-count').textContent = r30.length;

  const days = new Set(arsOnly.map(e => new Date(e.date).toDateString())).size || 1;

  const totalARS = arsOnly.reduce((s,e) => s+e.amount, 0);

  document.getElementById('dash-avg').textContent = '$' + fmt(totalARS / days);

  if (arsOnly.length) {

    const max = arsOnly.reduce((a,b) => a.amount > b.amount ? a : b);

    document.getElementById('dash-max').textContent = '$' + fmt(max.amount);

    document.getElementById('dash-max-desc').textContent = max.desc;

  }

  const catTotals = {};

  arsOnly.forEach(e => catTotals[e.cat] = (catTotals[e.cat]||0) + e.amount);

  const sorted = Object.entries(catTotals).sort((a,b) => b[1]-a[1]);

  if (sorted.length) {

    document.getElementById('dash-top-cat').textContent = CAT_EMOJI[sorted[0][0]] + ' ' + sorted[0][0];

    document.getElementById('dash-top-cat-sub').textContent = '$' + fmt(sorted[0][1]);

  }

  const maxCat = sorted.length ? sorted[0][1] : 1;

  document.getElementById('cat-breakdown').innerHTML = sorted.slice(0,8).map(([cat, val]) => `

    <div class="cat-row">

      <span class="cat-emoji">${CAT_EMOJI[cat]||'📦'}</span>

      <span class="cat-name">${cat}</span>

      <div class="cat-bar-wrap"><div class="cat-bar" style="width:${Math.round(val/maxCat*100)}%"></div></div>

      <span class="cat-val">$${fmt(val)}</span>

    </div>

  `).join('') || '<div style="color:var(--text3);font-size:13px;">Sin datos ARS en los últimos 30 días.</div>';

  renderEvolChart();

}


function renderEvolChart() {

  const now = new Date(), labels = [], data = [];

  for (let i = evolPeriod - 1; i >= 0; i--) {

    const d = new Date(now); d.setDate(now.getDate() - i);

    labels.push(d.toLocaleDateString('es-AR', {day:'numeric',month:'short'}));

    const total = expenses.filter(e => e.currency === 'ARS' && new Date(e.date).toDateString() === d.toDateString()).reduce((s,e) => s+e.amount, 0);

    data.push(Math.round(total));

  }

  if (evolChart) { evolChart.destroy(); evolChart = null; }

  const ctx = document.getElementById('chart-evol').getContext('2d');

  const grad = ctx.createLinearGradient(0, 0, 0, 220);

  grad.addColorStop(0, 'rgba(108,99,255,0.35)'); grad.addColorStop(1, 'rgba(108,99,255,0.02)');

  evolChart = new Chart(ctx, {

    type: 'line',

    data: { labels, datasets: [{ data, fill: true, backgroundColor: grad, borderColor: '#6C63FF', borderWidth: 2, pointBackgroundColor: '#6C63FF', pointRadius: 3, pointHoverRadius: 5, tension: 0.4 }] },

    options: {

      responsive: true, maintainAspectRatio: false,

      plugins: { legend: { display: false }, tooltip: { backgroundColor: '#2D2D2D', titleColor: '#A0A0A0', bodyColor: '#F5F5F5', borderColor: 'rgba(255,255,255,0.1)', borderWidth: 1, callbacks: { label: ctx => ' $' + ctx.parsed.y.toLocaleString('es-AR') } } },

      scales: {

        x: { ticks: { color: '#636363', font: { size: 11 }, maxRotation: 0 }, grid: { color: 'rgba(255,255,255,0.04)' } },

        y: { ticks: { color: '#636363', font: { size: 11 }, callback: v => '$' + (v >= 1000 ? Math.round(v/1000)+'k' : v) }, grid: { color: 'rgba(255,255,255,0.04)' }, beginAtZero: true }

      }

    }

  });

}
